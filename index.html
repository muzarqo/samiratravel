// ---- File picker: kompres gambar & simpan sebagai base64 ---- let buktiDataUrl = null; // data:image/jpeg;base64,.... let buktiFileName = ''; function compressImage(file, maxWidth = 1280, quality = 0.72){ return new Promise((resolve, reject) => { const reader = new FileReader(); reader.onload = (e) => { const img = new Image(); img.onload = () => { let { width, height } = img; if (width > maxWidth){ height = Math.round(height * (maxWidth / width)); width = maxWidth; } const canvas = document.createElement('canvas'); canvas.width = width; canvas.height = height; canvas.getContext('2d').drawImage(img, 0, 0, width, height); resolve(canvas.toDataURL('image/jpeg', quality)); }; img.onerror = reject; img.src = e.target.result; }; reader.onerror = reject; reader.readAsDataURL(file); }); } const fileDrop = document.getElementById('fileDrop'); const fileInput = document.getElementById('fileInput'); fileDrop.addEventListener('click', () => fileInput.click());
